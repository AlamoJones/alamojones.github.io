    #!/bin/bash
DIR="$HOME/dev/snowday/out"
n=0
while true; do
TIME=$(date +%d_%H%M%S)
printf -v n_pad "%05d" $n
rpicam-still -o "$DIR/frame_${n_pad}_${TIME}.jpg" --nopreview -t 1 --quality 100 --denoise off --raw
let n++
sleep 30
done
